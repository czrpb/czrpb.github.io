<script>
(function(){
  const w=document.getElementById('timer-widget'),
        d=document.getElementById('timer-display'),
        c=document.getElementById('timer-controls'),
        b=document.getElementById('timer-btn'),
        mIn=document.getElementById('timer-min'),
        sIn=document.getElementById('timer-sec');
  let iv=null, left=0, running=false;

  w.onmouseenter=()=>{ c.style.opacity='1'; d.style.opacity='.15'; };
  w.onmouseleave=()=>{ c.style.opacity='0'; d.style.opacity='1'; };

  function fmt(s){ return String(Math.floor(s/60)).padStart(2,'0')+':'+String(s%60).padStart(2,'0'); }

  function tick(){
    if(left<=0){ clearInterval(iv); iv=null; running=false; b.textContent='GO';
      d.style.color='#f44'; d.textContent='00:00'; return; }
    left--; d.textContent=fmt(left);
    if(left<=10) d.style.color='#f44';
  }

  b.onclick=()=>{
    if(running){ clearInterval(iv); iv=null; running=false; b.textContent='GO'; return; }
    left=(parseInt(mIn.value)||0)*60+(parseInt(sIn.value)||0);
    if(left<=0) return;
    d.style.color='#0f0'; d.textContent=fmt(left);
    running=true; b.textContent='â– ';
    iv=setInterval(tick,1000);
  };
})();
</script>
